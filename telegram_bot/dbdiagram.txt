Table users {
  id integer [primary key]
  username varchar
  role varchar
  created_at timestamp
}

Table film {
  id integer [primary key]
  name varchar
  film file
  genre_id integer [primary key]
  actor_id integer [primary key]
  description text
  release_date date
  poster_url varchar
  duration integer
  rating varchar
}

Table genre {
  id integer [primary key]
  name varchar
}

Table actor {
  id integer [primary key]
  film_id integer [primary key]
  name varchar
  birthdate date
}

Table rating {
  id integer [primary key]
  film_id integer [primary key]
  user_id integer [primary key]
  rating_value float
  created_at timestamp
}

Table comment {
  id integer [primary key]
  film_id integer [primary key]
  user_id integer [primary key]
  comment_text text
  created_at timestamp
}

Table news {
  id integer [primary key]
  user_id integer [primary key]
  title varchar
  content text
  created_at timestamp
}

Table employees {
  id integer [primary key]
  user_id integer [primary key]
  department_id integer [primary key]
  position varchar
  salary decimal(10, 2)
  hire_date date
}

Table department {
  id integer [primary key]
  name varchar
  location varchar
}

Ref: comment.user_id > users.id // Отношение "многое к одному": много комментариев принадлежит одному пользователю
Ref: news.user_id > users.id // Отношение "многое к одному": много новостей принадлежит одному пользователю
Ref: film.genre_id > genre.id 
Ref: film.actor_id > actor.id
